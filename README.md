```bash
npm install --save ts-ship-table
```

```tsx
import React, { Component } from 'react'

import MyComponent from 'ts-ship-table'
import 'ts-ship-table/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
